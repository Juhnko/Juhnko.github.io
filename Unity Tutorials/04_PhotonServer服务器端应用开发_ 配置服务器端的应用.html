Log4Net插件的使用

参考官方文档https://logging.apache.org/log4net/
01、添加引用，引入log4net.dll和ExitGames.Logging.Log4Net.dll两个文件 。（两个文件要从src-server文件夹下获取）服务端开发的SDK。   SDK：Software develop kit  软件开发工具

02、配置log4net的配置文件，log4net.config
         配置文件——始终复制；会在程序的打包输出目录找到这个配置文件。
<?xml version="1.0" encoding="utf-8" ?>
<log4net debug="false" update="Overwrite">
<appender name="RollingFileAppender" type="log4net.Appender.RollingFileAppender">
<file type="log4net.Util.PatternString" value="%property{Photon:ApplicationLogPath}\\2021LJKGameServer.log" /> <!-- log文件的输出目录 -->
<appendToFile value="true" />
<maximumFileSize value="5000KB" />
<maxSizeRollBackups value="2" />
<layout type="log4net.Layout.PatternLayout">
<conversionPattern value="%d [%t] %-5p %c - %m%n" />
</layout>
</appender>
<appender name="ConsoleAppender" type="log4net.Appender.ConsoleAppender">
<layout type="log4net.Layout.PatternLayout">
<param name="ConversionPattern" value="%d [%t] %-5p %c - %m%n" />
</layout>
<filter type="log4net.Filter.LevelRangeFilter">
<levelMin value="DEBUG" />
<levelMax value="FATAL" />
</filter>
</appender>
<!-- logger -->
<root>
<level value="INFO" />
<!--<appender-ref ref="ConsoleAppender" />-->
<appender-ref ref="RollingFileAppender" />
</root>
<logger name="OperationData">
<level value="INFO" />
</logger>
</log4net>
03、在程序中进行日志的初始化
using System.IO;
using ExitGames.Logging;
using ExitGames.Logging.Log4Net;
using log4net.Config;
using Photon.SocketServer;
namespace LJKGameServer
{
public class LjkGameServer:ApplicationBase
{
private static readonly ILogger log = LogManager.GetCurrentClassLogger();
//当一个客户端请求链接的时候
//我们要使用PeerBase,表示和一个客户端的链接
//PhotonServer自动调用
protected override PeerBase CreatePeer(InitRequest initRequest)
{
log.Info("一个客户端链接成功了！！!");
return new ClientPeer(initRequest);
}
//初始化
protected override void Setup()
{
//日志的初始化
//01、log4net日志输出的目录
log4net.GlobalContext.Properties["Photon:ApplicationLogPath"] = Path.Combine(Path.Combine(this.ApplicationRootPath,"bin_win64"),"log");//程序根目录下的log目录下
//02、log4net配置文件的位置
FileInfo configFileInfo =new FileInfo(Path.Combine(this.BinaryPath,"log4net.config"));//this.BinaryPath：获取部署目录。Path.Combine连接完整的目录名
if (configFileInfo.Exists)
{
LogManager.SetLoggerFactory(Log4NetLoggerFactory.Instance);//指定使用Log4Net为日志文件插件
XmlConfigurator.ConfigureAndWatch(configFileInfo);//让Log4Net读取配置文件
}
log.Info("Setup Completed !!!");
}
//Server端关闭的时候
protected override void TearDown()
{
log.Info("一个服务器应用关闭链接了！！!");
}
}
