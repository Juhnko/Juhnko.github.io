using System.IO;
using ExitGames.Logging;
using ExitGames.Logging.Log4Net;
using log4net.Config;
using Photon.SocketServer;
namespace LJKGameServer
{
public class LjkGameServer:ApplicationBase
{
//创建日志
public static readonly ILogger log = LogManager.GetCurrentClassLogger();//ExitGames.Logging命名空间下的
//监听客户端的请求，并创建一个客户端的连接peer
//当一个客户端请求链接的时候
//我们要使用PeerBase,表示和一个客户端的链接
//PhotonServer自动调用
protected override PeerBase CreatePeer(InitRequest initRequest)
{
log.Info("一个客户端链接成功了！！!");
return new ClientPeer01(initRequest);
}
//初始化，对日志配置及输出。以后根据日志打印情况，参考服务端与客户端的通信状态
protected override void Setup()
{
//---日志的初始化---
//01、log4net日志——-输出的目录
log4net.GlobalContext.Properties["Photon:ApplicationLogPath"] = Path.Combine(Path.Combine(this.ApplicationRootPath,"bin_win64"),"log");//程序根目录下的log目录下
//02、log4net————配置文件的目录
FileInfo configFileInfo =new FileInfo(Path.Combine(this.BinaryPath,"log4net.config"));//this.BinaryPath：获取部署目录。Path.Combine连接完整的目录名
if (configFileInfo.Exists)
{
LogManager.SetLoggerFactory(Log4NetLoggerFactory.Instance);//指定使用Log4Net为日志文件插件
XmlConfigurator.ConfigureAndWatch(configFileInfo);//让Log4Net读取配置文件
}
//03、日志输出信息
log.Info("Setup Completed !!!");
}
//Server端关闭的时候
protected override void TearDown()
{
log.Info("一个服务器应用关闭链接了！！!");
}
}
}
04、创建服务端对ClientPeer01的监听
using System.Collections.Generic;
using Photon.SocketServer;
using PhotonHostRuntimeInterfaces;
namespace LJKGameServer
{
public class ClientPeer01:ClientPeer
{
public ClientPeer01(InitRequest initRequest) : base(initRequest)
{
}
//处理客户端ClientPeer01的请求
protected override void OnOperationRequest(OperationRequest operationRequest, SendParameters sendParameters)
{
switch (operationRequest.OperationCode)//通过OperationCode区分请求
{
case 1:
LjkGameServer.log.Info("收到了一个客户端的请求！！！");
//请求
Dictionary<byte, object> data = operationRequest.Parameters;
object intValue;
data.TryGetValue(1, out intValue);
object stringValue;
data.TryGetValue(2, out stringValue);
string s = string.Format("得到的数据是1：{0}，2：{1}", intValue.ToString(), stringValue.ToString());
LjkGameServer.log.Info(s);//将请求信息显示在服务端的log日志中
//响应(服务端也可以不做响应)
OperationResponse opResponse = new OperationResponse(1); // OperationCode：“1”
Dictionary<byte, object> data02 = new Dictionary<byte, object>();
data02.Add(1, 555666);
data02.Add(2, "dk000可可");
opResponse.SetParameters(data02);
SendOperationResponse(opResponse,sendParameters);//给客户端一个响应，客户端接收到后在客户端现实出来
break;
//发送广播事件（可以在服务器的任何地方向客户端发起SendEvent）
EventData eventData=new EventData(1);
Dictionary<byte, object> data03 = new Dictionary<byte, object>();
data03.Add(1, 99998888);
data03.Add(2, "dk111222");
eventData.Parameters = data03;
SendEvent(eventData,new SendParameters());
default:
break;
}
}
//处理客户端断开链接的后续工作
protected override void OnDisconnect(DisconnectReason reasonCode, string reasonDetail)
{
}
}
