03、明确不同文件夹的用处： 
  • Editor：Editor文件夹可以在根目录下，也可以在子目录里，只要名子叫Editor就可以，Editor下面放的所有资源文件或者脚本文件都不会被打进发布包中，并且脚本也只能在编辑时使用。一般呢会把一些工具类的脚本放在这里，或者是一些编辑时用的DLL。 比如我们现在要做类似技能编辑器，那么编辑器的代码放在这里是再好不过了，因为实际运行时我们只需要编辑器生成的文件，而不需要编辑器的核心代码。
  • Editor Default Resources：Editor Default Resources注意中间是有空格的，它必须放在Project视图的根目录下，如果你想放在/xxx/xxx/Editor Default Resources 这样是不行的。你可以把编辑器用到的一些资源放在这里，比如图片、文本文件、等等。它和Editor文件夹一样都不会被打到最终发布包里，仅仅用于开发时使用。你可以直接通过EditorGUIUtility.Load去读取该文件夹下的资源。
  • Plugins：如果做手机游戏开发一般 andoird 或者 ios 要接一些sdk 可以把sdk依赖的库文件 放在这里，比如 .so .jar .a 文件。这样打完包以后就会自动把这些文件打在你的包中。
  • Resources：可以在根目录下，也可以在子目录里，只要名子叫 Resources 就可以。Resources文件夹下的资源不管你用还是不用都会被打包进.apk或者.ipa。
  • StreamingAssets：这个文件夹下的资源也会全都打包在.apk或者.ipa 它和Resources的区别是，Resources会压缩文件，但是它不会压缩原封不动的打包进去。并且它是一个只读的文件夹，就是程序运行时只能读 不能写。它在各个平台下的路径是不同的，不过你可以用Application.streamingAssetsPath 它会根据当前的平台选择对应的路径。

二、UI框架的搭建

01、为什么要封装UI框架？
        方便管理、提高代码的可读性
02、功能
  •      启动关闭
  •      按钮点击注册
  •      节点注册

03、规定
  •  所有UIPrefab都要放在UIPrefabs文件夹下
  •  UIPrefab的名称要与对应的类名保持一致
  •  所有UIRrefab的根节点都是全屏
UI框架代码
using System.Collections.Generic;
using UnityEngine;
public class UIManager : UIPanelBase
{
public const string UIPath = "UIPrefabs/";
public static GameObject mUIRoot;
public static List<UIPanelBase> mUIList = new List<UIPanelBase>();
private static void InitData()
{
mUIRoot = GameObject.Find("Canvas");
}
//进入接口
public static T OnEnterUI<T>() where T:UIPanelBase
{
//01、Prefab加载
//02、实例化Prefab
//03、界面初始化
//04、显示UI成功
//... ...
//05、关闭
if (mUIRoot==null) {InitData();}
GameObject go = Resources.Load<GameObject>(UIPath + typeof(T).ToString());//t的类型
GameObject ui = Instantiate(go,mUIRoot.transform,false);
ui.transform.localPosition=Vector3.zero;
ui.transform.localRotation=Quaternion.identity;
ui.transform.localScale=Vector3.one;
RectTransform rect = ui.GetComponent<RectTransform>();
rect.offsetMin = Vector2.zero;
rect.offsetMax = Vector2.zero;
rect.anchorMin=Vector2.zero;
rect.anchorMax=Vector2.one;
T t = ui.AddComponent<T>();
t.__init_node(t.transform);
t.OnNodeLoad();
mUIList.Add(t);
return t;
}
//退出接口
public static void OnExitUI(UIPanelBase uiPanelBase)
{
mUIList.Remove(uiPanelBase);
Destroy(uiPanelBase.gameObject);
}
//退出所有
public static void OnExitAllUI()
{
foreach (var ui in mUIList)
{
Destroy(ui.gameObject);
}
mUIList.Clear();
}
}
using System;
using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.UI;
public class UIPanelBase : MonoBehaviour
{
//开始
private void Start()
{
OnEnter();
}
//销毁
private void OnDestroy()
{
OnExit();
}
//关闭
private void OnClose()
{
UIManager.OnExitUI(this);
}
//递归函数，通过递归函数获取父类及其子类所有Button组件来注册点击事件
public void __init_node(Transform transform)
{
//查看父对象
OnNodeAsset(transform.name, transform.gameObject);
Button button = transform.GetComponent<Button>();//尝试获取button组件
if (button != null)
{
button.onClick.AddListener(() => OnButtonClick(button.name, button.gameObject));
}
for (int i = 0; i < transform.childCount; i++)
{
//查看子对象
__init_node(transform.GetChild(i));
}
}
//UI加载的时候
public virtual void OnNodeLoad()
{
}
//UI启动完成
public virtual void OnEnter()
{
Debug.Log("UI启动完成！！");
}
public virtual void OnExit()
{
}
public virtual void OnButtonClick(string btnName, GameObject gameObject)
{
}
public virtual void OnNodeAsset(string name, GameObject gameObject)
{
}
}
using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.UI;
public class UI_MainPanel : UIPanelBase
{
[SerializeField]
private Text mTextTitle;
[SerializeField]
private Button mBtnStart;
[SerializeField]
private Button mBtnClose;
[SerializeField]
private Button mBtnShow;
[SerializeField]
private Button mBtnCloseAll;
public override void OnNodeLoad()
{
mTextTitle.text = "LJK_3DTowerDefenceDemo";
mTextTitle.transform.localPosition = new Vector3(Random.Range(-300, 300),Random.Range(-300, 300));
Debug.Log("OnNodeLoad");
}
//获取标题
public override void OnNodeAsset(string name, GameObject go)
{
switch (name)
{
case "text_title":
mTextTitle = go.GetComponent<Text>();
break;
case "btn_start":
mBtnStart= go.GetComponent<Button>();
break;
}
}
public override void OnButtonClick(string btnName, GameObject gameObject)
{
switch (btnName)
{
case "btn_start":
Debug.Log("游戏加载进来了！！！");
break;
case "btn_close" :
UIManager.OnExitUI(this);
Debug.Log("游戏已经关闭了！！！");
break;
case "btn_show":
UIManager.OnEnterUI<UI_MainPanel>();
Debug.Log("游戏已经又启动了！！！");
break;
case "btn_closeAll":
UIManager.OnExitAllUI();
Debug.Log("游戏界面全部关闭了！！！");
break;
}
}
public override void OnEnter()
{
base.OnEnter();
Debug.Log("OnEnter");
}
public override void OnExit()
{
base.OnExit();
Debug.Log("OnExit");
}
}
