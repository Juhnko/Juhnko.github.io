DoTween笔记

入门理解：补间动画

public Transform gameObject;
public Vector3 myValue;
// Start is called before the first frame update
void Start()
{
Vector3 myPos = gameObject.position;
myValue =new Vector3(myPos.x,myPos.y,myPos.z) ;
//对变量动画（通过差值的方式修改一个值的变化，渐变过去）
DOTween.To(() => myValue, x => myValue = x, new Vector3(10,10,10),2);
}
// Update is called once per frame
void Update()
{
gameObject.position = myValue;//每一帧的值与myValue的值保持一样。
}

02、使用可视化界面的方式调用DoTween

private DOTweenAnimation tweenAnimation;
private bool isOnDisplay;
// Start is called before the first frame update
void Start()
{
tweenAnimation = GetComponent<DOTweenAnimation>();
}
public void OnMyClickAnimation()
{
isOnDisplay = !isOnDisplay;
if (isOnDisplay)
{
tweenAnimation.DOPlayForward();
}
else
{
tweenAnimation.DOPlayBackwards();
}
}
