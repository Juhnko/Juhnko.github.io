二、Manager文件夹
     UIManager（UI框架的核心管理类）
    1、解析保存所有的面板信息（PanelPathDict）    路径信息的保存
    2、创建保存所有面板的实例（PanelDict）                               实例BasePanel对象
    3、管理保存所有显示的面板  （所有显示在界面上的面板，用栈管理，用栈保存显示的面板，通过栈的出栈入栈，控制页面的显示/隐藏）Stack
          面板的三种状态：prefab状态、显示状态、隐藏状态。
          使用“显示页面的容器栈”（PanelStack）：先进后出，后进先出。（数据结构）
    4、给页面（Panel）添加生命周期（页面状态流程）
          页面的状态：页面进入（OnEnter），页面暂停（OnPause），页面继续（OnResume），页面移出（OnExit）。
          页面进入（OnEnter）      ：   界面显示出来。
          页面暂停（OnPause）     ：   界面暂停（弹出了其他界面）
          页面继续（OnResume）  ：   界面继续（其他界面移除，恢复本界面交互）
          页面移出（OnExit）         ：   界面被移除了（从页面移除，不显示）
    5、任务栈Stack
          panlStack.Peek();
          panlStack.Pop();
          任务栈只能跟栈顶的任务做交互。
Stack类
  • Push将元素插入Stack的顶部。  插入
  • Peek返回位于Stack 顶部的元素（但不会将其从Stack 中删除） 返回
  • Pop从Stack 的顶部移除元素。  移除

/// <summary>
/// 把某个页面插入到栈 （把某个页面显示在界面上）
/// </summary>
public void PushPanel(UIPanelType uiPanelType)//显示到界面上
{
if (panelStack==null)
{
panelStack = new Stack<BasePanel>();//创建栈，作为页面管理容器
}
//判断一下栈里面是否有页面
if (panelStack.Count>0)
{
//先让已存在的（上一个）panel暂停
BasePanel topPanel = panelStack.Peek();//返回，返回位于Stack顶部的元素。 TODO
topPanel.OnPause();
}
//创建本次的uiPanelType，并保存在栈容器中
BasePanel panel = GetPanel(uiPanelType);//--------得到面板实例化出来
panel.OnEnter();
panelStack.Push(panel);//-----------插入，将元素插入Stack的顶部。（实例化出来之后，将面板添加到栈里面）
}
/// <summary>
/// 把某个页面移出栈（把某个页面从界面上移除）
/// </summary>
public void PopPanel()//从界面上移除
{
if (panelStack==null) panelStack = new Stack<BasePanel>();
if (panelStack.Count<=0) return;
//关闭栈顶页面的显示
BasePanel topPanel = panelStack.Pop();//移除 并找到 栈顶的panel
topPanel.OnExit();//把topPanel移除
if (panelStack.Count<=0) return;
BasePanel topPanel02 = panelStack.Peek();//查看栈顶的panel
topPanel02.OnResume();
}

#region 将Json保存所有面板的Prefab路径保存到panelPathDict集合中
private void ParseUIPanelJson()
{
panelPathDict = new Dictionary<UIPanelType, string>();
TextAsset textAsset = Resources.Load<TextAsset>("UIPanelType");//加载json文件
UIPanelTypeJson jsonObject = JsonUtility.FromJson<UIPanelTypeJson>(textAsset.text);//将json转成对象
foreach (var uiPanelInfo in jsonObject.PanelInfoList )
{
panelPathDict.Add(uiPanelInfo.uiPanelType, uiPanelInfo.path);//存放在panelPathDict集合中，存放预置体和相对应的路径
}
}

四、拓展工具类Dictionary文件
public static class DictionaryExtension
{
//对字典类的拓展（给字典类拓展方法）
/// <summary>
/// 尝试根据Key，得到Value。
/// 得到了的话返回Vaulue,没有得到的话返回null。
/// </summary>
/// <param name="dictionary">字典对象</param>
/// <param name="key">传入的键</param>
/// <typeparam name="TKey">字典的键</typeparam>
/// <typeparam name="TValue">字典的值</typeparam>
public static TValue TryGet<TKey,TValue>(this Dictionary<TKey,TValue> dictionary,TKey key )
{
dictionary.TryGetValue(key, out var value);
return value;
}
}

02、KnapsackPanel类，来控制“KnapsackPanel界面”
public class KnapsackPanel : BasePanel
{
private CanvasGroup _canvasGroup;
private void Start()
{
if (_canvasGroup==null) _canvasGroup = GetComponent<CanvasGroup>();
}
public override void OnEnter()
{
if (_canvasGroup==null) _canvasGroup = GetComponent<CanvasGroup>();
_canvasGroup.alpha = 1f;
_canvasGroup.interactable = true;
Vector3 temp = transform.localPosition;
temp.x = 1000f;
transform.localPosition = temp;
transform.DOLocalMoveX(0,0.5f);
}
public override void OnExit()
{
// _canvasGroup.alpha = 0f;
_canvasGroup.interactable = false;
transform.DOLocalMoveX(1200f, 0.5f).OnComplete(() => _canvasGroup.alpha = 0f); //播放完执行回调函数
}
public override void OnPause()
{
_canvasGroup.interactable = false;
}
public override void OnResume()
{
_canvasGroup.interactable = true;
}
public void OnClosePanel()
{
UIManager.Instance.PopPanel();
}
