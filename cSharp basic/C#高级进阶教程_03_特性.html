特性初步了解：
      1.  中括号声明，每一个特性都可以带来对应的功能。
      2. 影响编译器的运行，影响程序运行。
      3. 就是一个类，继承自attribute。（直接/间接继承都可以）
      4. 一般以Attribute结尾，声明时可以省略掉。
实际上，特性添加后，编译会在元素的内部产生IL,但是我们是没有办法直接使用的，而且在metadata（元数据）里面会有记录。
[AttributeUsage(AttributeTargets.All,AllowMultiple = true)] //AttributeUsage用来修饰特性的特性
//[AttributeUsage(AttributeTargets.All, AllowMultiple = false, Inherited = true)] //Inherited表示是否可以被继承，默认是true。AllowMultiple 运行多重使用的默认值为false。
public class CustomAttribute : Attribute
{
public CustomAttribute()
{ }
public CustomAttribute(int id)
{
Console.WriteLine("********************");
}
public string Description { get; set; }
public string Remark = null;
public void Show()
{
Console.WriteLine($"This is {nameof(CustomAttribute)}");
}
}
[Custom] //特性无参数的构造函数
[Custom()] //同样也是特性的无参数的构造函数
[Custom(123), Custom(123, Description = "1234")] //两个特性修饰
[Custom(123, Description = "1234", Remark = "2345")] //其实在实例化一个类，方法不能传入
[CustomAttribute] //特性也可以修饰方法
public void Study()

[Custom()]
[return: Custom()] //特性也可以修饰返回值
public string Answer([Custom]string name) //特性也可以修饰参数
{
return $"This is {name}";
}
以上为特性的语法。
特性有什么用？特性有什么价值？
      特性：是用于在运行时传递程序中的各种元素（比如类、方法、结构、枚举、组建等）的行为信息的声明性标签。
                 可以通过使用特性想程序添加声明性信息。
                 一个声明性标签是通过放置在它所应用的元素前面的方括号（[]）来描述的。
      特性作用：用于添加元数据，如编译器指令和注释、描述、方法、类等其他信息。
       .Net框架提供了两种类型的特性：预定义特性和自定义特性。

      预定义特性：
              AttributeUsage  修饰特性的特性
              Conditional          这个预定义特性标记了一个条件方法，其执行依赖于指定的预处理标识符。
              Obsolete               标记了不应被使用的程序实体。（过时的）
      
      在程序运行的过程中能找到特性。  
      Unity中特性的应用！
      特性类似于类的补充。增加类额外的信息和行为！（没有破坏类的封装的前提下，没有修改类）

 01. 通过反射获取类型，检查该类型有没有被特性（CustomAttribute）修饰。
Type type = typeof(Student); //student.GetType(); //01.通过反射获取类型
if (type.IsDefined(typeof(CustomAttribute), true)) //type.IsDefined检查有没有 __性能高相对于（type.GetCustomAttribute）

attribute.Show(); //03.通过实例调用其方法
}

[Remark("删除")]

public class RemarkAttribute : Attribute
{
private string _Remark = null;
public RemarkAttribute(string remark) //构造函数
{
this._Remark = remark;
}
public string GetRemark() //获取方法
{
return this._Remark;
}
}
public static class RemarkExtension
{
public static string GetRemark(this Enum value) // 扩展方法的使用：
//01.静态类的静态方法
//02.同一命名空间下，Enum类型的调用此方法。
//03.this关键字的使用
{
Type type = value.GetType(); // 通过反射获取类型
FieldInfo field = type.GetField(value.ToString()); // 通过反射获取到枚举类型的字段(MyAttribute.Extension(命名空间).UserState（枚举） Normal（字符串信息）)
if (field.IsDefined(typeof(RemarkAttribute), true)) // 检查枚举类型的字段有没有被特性修饰
{
//利用反射通过字段获取到修饰字段的特性
RemarkAttribute attribute = (RemarkAttribute)field.GetCustomAttribute(typeof(RemarkAttribute), true);
return attribute.GetRemark(); // 返回特性中的方法
}
else
{
return value.ToString(); // 返回枚举类型转成字符串的值
}
}
}

public abstract class AbstractValidateAttribute : Attribute //抽象类
{
public abstract bool Validate(object value);
}
public class LongAttribute : AbstractValidateAttribute
{
private long _Min = 0;
private long _Max = 0;
public LongAttribute(long min, long max)
{
this._Min = min;
this._Max = max;
}
public override bool Validate(object value)//" " //重写方法
{
//可以改成一句话
if (value != null && !string.IsNullOrWhiteSpace(value.ToString())) //值不能为空，ToString（）转成字符串后也不能为空。
{
if (long.TryParse(value.ToString(), out long lResult)) //if((,out ))的使用，//如果成功的话，将值赋给lResult，如果不成功则不进行里面的操作。
{
if (lResult > this._Min && lResult < this._Max)
{
return true;
}
}
}
return false; //任何不对，都返回false
}

public static class ValidateExtension
{
//校验拓展类，最好后期使用泛型，目前使用的是object
public static bool Validate(this object oObject)
{
Type type = oObject.GetType(); // 通过反射，找到类型
foreach (var prop in type.GetProperties()) // 通过反射，获取类型的全部属性
{
//遍历所有属性是有LongAttribute特性修饰
if (prop.IsDefined(typeof(LongAttribute), true)) // 通过反射，检查属性是否有自定义特性LongAttribute修饰
{
LongAttribute attribute = (LongAttribute)prop.GetCustomAttribute(typeof(LongAttribute), true);
if (!attribute.Validate(prop.GetValue(oObject)))
{
return false;
}
}
//遍历所有属性是有LengAttribute特性修饰
if (prop.IsDefined(typeof(LengAttribute), true))
{
LengAttribute attribute = (LengAttribute)prop.GetCustomAttribute(typeof(LengAttribute), true);
if (!attribute.Validate(prop.GetValue(oObject)))
{
return false;
}
}
}
return true; // 如果有LongAttribute或LengAttribute修饰，则返回true
}
}

public abstract class AbstractValidateAttribute : Attribute //抽象类
{
public abstract bool Validate(object value);
}
[AttributeUsage(AttributeTargets.Property | AttributeTargets.Field)] //自定义特性的特性修饰
public class LengAttribute : AbstractValidateAttribute //继承抽象类，重写方法，来具体实现
{
private int _Min = 0;
private int _Max = 0;
public LengAttribute(int min, int max)
{
this._Min = min;
this._Max = max;
}
public override bool Validate(object value)//" "
{
if (value != null && !string.IsNullOrWhiteSpace(value.ToString()))
{
int length = value.ToString().Length;
if (length > this._Min && length < this._Max)
{
return true;
}
}
return false;
}
}
public class LongAttribute : AbstractValidateAttribute //继承抽象类，重写方法，来具体实现
{
private long _Min = 0;
private long _Max = 0;
public LongAttribute(long min, long max)
{
this._Min = min;
this._Max = max;
}
public override bool Validate(object value)//" "
{
//可以改成一句话
if (value != null && !string.IsNullOrWhiteSpace(value.ToString()))
{
if (long.TryParse(value.ToString(), out long lResult)) //如果成功的话，将值赋给lResult，如果不成功则不进行里面的操作

if (lResult > this._Min && lResult < this._Max)
{
return true;
}

public static class ValidateExtension //改造后的验证拓展类，只要验证的特性基于AbstractValidateAttribute（抽象验证特性类），那么这个ValidateExtension则不需要更改了。
{
//校验拓展类，最好后期使用泛型，目前使用的是object
public static bool Validate(this object oObject)
{
Type type = oObject.GetType(); //找到类型
foreach (var prop in type.GetProperties()) //获取类型的全部属性
{
if (prop.IsDefined(typeof(AbstractValidateAttribute), true)) //校验属性是否有特性
{
object[] attributeArray = prop.GetCustomAttributes(typeof(AbstractValidateAttribute), true); //得到自定义的特性数组
foreach (AbstractValidateAttribute attribute in attributeArray) //将每个特性转成抽象的验证特性
{
if (!attribute.Validate(prop.GetValue(oObject))) //这个的使用？（直接又调用抽象特性的方法）
{
return false; //表示终止，逻辑思维：一旦遇到一个错误直接结束！
}
}
}
}
}
return true;
}
